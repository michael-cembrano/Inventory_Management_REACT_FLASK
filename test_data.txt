### Authentication Requests

```bash
# Login Request
curl -X POST http://localhost:5001/api/auth/login \
  -H "Content-Type: application/json" \
  -d '{
    "username": "admin",
    "password": "admin123"
  }'

# Response:
{
  "access_token": "eyJ0eXAiOiJKV1QiLCJhbGciOiJIUzI1NiJ9...",
  "user": {
    "username": "admin",
    "role": "admin",
    "email": "admin@inventory.com"
  }
}
```

### Inventory Management Requests

```bash
# Get All Inventory
curl -X GET http://localhost:5001/api/inventory \
  -H "Authorization: Bearer YOUR_TOKEN"

# Add New Inventory Item
curl -X POST http://localhost:5001/api/inventory \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "name": "New Wireless Keyboard",
    "category": "Electronics",
    "quantity": 50,
    "price": 79.99,
    "description": "Mechanical wireless keyboard with RGB lighting"
  }'

# Update Inventory Item
curl -X PUT http://localhost:5001/api/inventory/1 \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "name": "MacBook Pro 16-inch M3 (Updated)",
    "quantity": 10,
    "price": 2899.99
  }'
```

### Order Management Requests

```bash
# Create New Order
curl -X POST http://localhost:5001/api/orders \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "customer_name": "Test Customer",
    "items": [
      {
        "id": 1,
        "name": "MacBook Pro 16-inch M3",
        "quantity": 1,
        "price": 2999.99
      }
    ]
  }'

# Get All Orders
curl -X GET http://localhost:5001/api/orders \
  -H "Authorization: Bearer YOUR_TOKEN"
```

### Admin Requests

```bash
# Get System Stats (Admin Only)
curl -X GET http://localhost:5001/api/admin/system-stats \
  -H "Authorization: Bearer ADMIN_TOKEN"

# Create New User (Admin Only)
curl -X POST http://localhost:5001/api/admin/users \
  -H "Authorization: Bearer ADMIN_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "username": "newuser",
    "email": "newuser@inventory.com",
    "password": "newuser123",
    "role": "user"
  }'

# Backup Database (Admin Only)
curl -X POST http://localhost:5001/api/admin/backup \
  -H "Authorization: Bearer ADMIN_TOKEN"
```
